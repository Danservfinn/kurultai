## 3. Color Palette & Atmosphere

### 3.1 Terminal Color Mapping

```python
# ANSI Color Codes for the Horde Theme
HORDE_COLORS = {
    # Imperial Colors (Khan's commands)
    "imperial_gold": "\033[38;5;220m",      # Gold/Khaki
    "imperial_red": "\033[38;5;196m",       # Deep Red
    "imperial_border": "\033[38;5;172m",    # Orange accent

    # Horde Colors (Agents/Forces)
    "horde_blue": "\033[38;5;67m",         # Steel Blue
    "horde_cyan": "\033[38;5;45m",         # Cyan
    "horde_green": "\033[38;5;107m",       # Olive

    # Status Colors
    "success_gold": "\033[38;5;226m",      # Bright Gold
    "warning_orange": "\033[38;5;214m",    # Orange
    "danger_red": "\033[38;5;160m",        # Deep Red
    "info_gray": "\033[38;5;245m",         # Gray

    # Territory Colors (for progress maps)
    "territory_conquered": "\033[48;5;22m",      # Green background
    "territory_contested": "\033[48;5;214m",     # Orange background
    "territory_target": "\033[48;5;160m",        # Red background
    "territory_pending": "\033[48;5;245m",       # Gray background

    # Reset
    "reset": "\033[0m"
}
```

### 3.2 Color Usage Guide

```text
IMPERIAL (Gold/Red) - Use for:
  - The Khan's direct commands
  - Victory announcements
  - Critical decisions
  - Opening/Closing ceremonies

HORDE (Blue/Cyan/Green) - Use for:
  - Agent status updates
  - Progress reports
  - Force movements
  - Battle maps

STATUS (Gold/Orange/Red/Gray) - Use for:
  - Success: Gold highlights
  - Warnings: Orange accents
  - Failures: Red emphasis
  - Info: Gray for secondary text
```

## Implementation Notes

### Python Helper Functions

```python
from time import sleep
import sys

def type_out(text, speed=0.02, color=None):
    """Type out text character by character for dramatic effect"""
    for char in text:
        if color:
            print(f"{color}{char}\033[0m", end='', flush=True)
        else:
            print(char, end='', flush=True)
        sleep(speed)
    print()

def reveal_lines(lines, delay=0.3):
    """Reveal multiple lines with staggered timing"""
    for line in lines:
        print(line)
        sleep(delay)

def pulse_progressBar(width=40, cycles=3):
    """Animate a pulsing progress bar"""
    for _ in range(cycles):
        for i in range(width + 1):
            bar = '█' * i + '░' * (width - i)
            print(f'\r[{bar}]', end='', flush=True)
            sleep(0.05)
        for i in range(width, -1, -1):
            bar = '█' * i + '░' * (width - i)
            print(f'\r[{bar}]', end='', flush=True)
            sleep(0.05)
    print()

def render_border_box(content, border_style="imperial"):
    """Render content inside themed border"""
    borders = {
        "imperial": ("╔", "╗", "╚", "╝", "║", "═"),
        "royal": ("╔", "╗", "╚", "╝", "║", "═"),
        "horde": ("┏", "┓", "┗", "┛", "┃", "━"),
        "scout": ("▛", "▜", "▙", "▟", "┃", "▀"),
    }
    tl, tr, bl, br, v, h = borders[border_style]
    width = max(len(line) for line in content) + 4

    print(tl + h * width + tr)
    for line in content:
        padding = ' ' * (width - len(line) - 2)
        print(v + ' ' + line + padding + ' ' + v)
    print(bl + h * width + br)
```
